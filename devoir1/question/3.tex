\section*{Question 3}
\begin{align*} 
\neg p \vee  (r \Rightarrow (\neg q)) &\equiv \neg p \vee \neg q \vee \neg r \\
\neg p \vee (r \Rightarrow (\neg q)) &\equiv \neg p \vee \neg(\neg(r \Rightarrow (\neg q)) \\
&\equiv \neg p \vee \neg(r \wedge \neg(\neg q)) \\
&\equiv \neg p \vee \neg r \vee \neg q
\end{align*}
